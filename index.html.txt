<script>
  // ===================== CONFIG =====================
  // Replace with your Render backend URL
  const API_BASE = "https://YOUR-RENDER-BACKEND.onrender.com";
  const ANALYZE_URL = API_BASE + "/analyze";

  // ===================== STATE =====================
  let DEMO_MODE = false;
  let lastPayload = null;

  const SAMPLES = {
    1: "java.lang.NullPointerException at com.company.billing.InvoiceService.processInvoice(InvoiceService.java:214)",
    2: "java.lang.OutOfMemoryError: Java heap space during batch job recon-nightly",
    3: "java.net.SocketTimeoutException: Read timed out calling https://upstream.api/orders"
  };

  function loadSample(n){
    document.getElementById("inp").value = SAMPLES[n] || "";
    document.getElementById("status").textContent = "Status: Sample loaded";
  }

  function applyDemoMode(){
    DEMO_MODE = !!document.getElementById("demoToggle").checked;

    if(lastPayload){
      renderSummary(lastPayload);
      renderDetails(lastPayload);
    }

    document.getElementById("status").textContent = DEMO_MODE
      ? "Status: Client Demo Mode ON (redacted)"
      : "Status: Client Demo Mode OFF";
  }

  // ===================== TABS =====================
  function switchTab(which){
    document.getElementById("tabSummary").classList.toggle("active", which==="summary");
    document.getElementById("tabDetails").classList.toggle("active", which==="details");
    document.getElementById("panelSummary").classList.toggle("active", which==="summary");
    document.getElementById("panelDetails").classList.toggle("active", which==="details");
  }

  function openDetails(which){
    switchTab("details");
    // we keep it simple: details is one combined JSON output
    // can be upgraded to per-section scroll later if needed
  }

  // ===================== REDACTION (Client Demo Mode) =====================
  function redactText(s){
    if(!s) return s;
    s = String(s);

    // Email addresses
    s = s.replace(/[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}/gi, "[redacted-email]");

    // Log paths
    s = s.replace(/\/var\/log\/[A-Za-z0-9_\-./]+/g, "[redacted-log-path]");

    // Correlation IDs like CORR-...
    s = s.replace(/\bCORR-[A-Z0-9\-]+\b/g, "[redacted-correlation-id]");

    // Internal-ish domains
    s = s.replace(/\b[a-z0-9.-]*\.(local|corp|internal)\b/gi, "[redacted-host]");

    return s;
  }

  function maybeRedact(obj){
    if(!DEMO_MODE) return obj;

    const walk = (x) => {
      if(x === null || x === undefined) return x;
      if(typeof x === "string") return redactText(x);
      if(Array.isArray(x)) return x.map(walk);
      if(typeof x === "object"){
        const out = {};
        for(const k of Object.keys(x)){
          if(k === "excerpt") out[k] = "[redacted-excerpt]";
          else out[k] = walk(x[k]);
        }
        return out;
      }
      return x;
    };
    return walk(obj);
  }

  // ===================== UI HELPERS =====================
  function setBadge(el, found){
    el.classList.remove("ok","no");
    if(found){
      el.classList.add("ok");
      el.textContent = "Found";
    } else {
      el.classList.add("no");
      el.textContent = "Not Found";
    }
  }

  function fmtConf(x){
    if(x === null || x === undefined || x === "") return "";
    const n = Number(x);
    if(Number.isNaN(n)) return "";
    return "confidence=" + n.toFixed(3);
  }

  // ===================== RENDERERS =====================
  function renderSummary(payload){
    const p = maybeRedact(payload);

    const sn = p.servicenow || {};
    const ml = p.mail || {};
    const lg = p.server_logs || {};

    setBadge(document.getElementById("badgeSN"), !!sn.found);
    setBadge(document.getElementById("badgeMail"), !!ml.found);
    setBadge(document.getElementById("badgeLogs"), !!lg.found);

    document.getElementById("sumSN").textContent   = sn.summary || "";
    document.getElementById("sumMail").textContent = ml.summary || "";
    document.getElementById("sumLogs").textContent = lg.summary || "";

    // show confidence lines if present
    document.getElementById("confSN").textContent   = fmtConf(sn.confidence);
    document.getElementById("confMail").textContent = fmtConf(ml.confidence);
    document.getElementById("confLogs").textContent = fmtConf(lg.confidence);
  }

  function renderDetails(payload){
    const p = maybeRedact(payload);

    const out = DEMO_MODE
      ? { mode: "Client Demo Mode (redacted)", details: (p.details || {}) }
      : {
          servicenow: p.servicenow,
          mail: p.mail,
          server_logs: p.server_logs,
          fallback: p.fallback,
          details: p.details
        };

    document.getElementById("detailsOut").textContent = JSON.stringify(out, null, 2);
  }

  // ===================== MAIN ACTION =====================
  async function runAnalyze(){
    const status = document.getElementById("status");
    const text = (document.getElementById("inp").value || "").trim();
    if(!text){
      status.textContent = "Status: Please paste an error first.";
      return;
    }

    status.textContent = "Status: Analyzing…";

    try{
      const r = await fetch(ANALYZE_URL, {
        method:"POST",
        headers:{"Content-Type":"application/json"},
        body: JSON.stringify({ error_text: text })
      });

      if(!r.ok){
        const t = await r.text();
        throw new Error(t || ("HTTP " + r.status));
      }

      const data = await r.json();
      lastPayload = data;

      renderSummary(data);
      renderDetails(data);

      status.textContent = DEMO_MODE
        ? "Status: Done (Client Demo Mode ON)"
        : "Status: Done";
    }catch(e){
      status.textContent = "Status: Error";
      document.getElementById("detailsOut").textContent = String(e && e.message ? e.message : e);
    }
  }

  function clearAll(){
    document.getElementById("inp").value = "";
    document.getElementById("status").textContent = "Status: Idle";
    document.getElementById("sumSN").textContent = "Awaiting analysis.";
    document.getElementById("sumMail").textContent = "Awaiting analysis.";
    document.getElementById("sumLogs").textContent = "Awaiting analysis.";
    document.getElementById("confSN").textContent = "";
    document.getElementById("confMail").textContent = "";
    document.getElementById("confLogs").textContent = "";
    document.getElementById("detailsOut").textContent = "Run analysis to view details.";
    lastPayload = null;
    switchTab("summary");
  }
</script>
